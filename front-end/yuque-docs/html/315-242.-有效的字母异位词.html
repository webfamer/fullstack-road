2. 解题思路
（1）首先将字符串转换为数组
（2）将数组元素进行排序
（3）将排序好的数组元素转化为字符串，并比较是否相等

/**
* @param {string} s
* @param {string} t
* @return {boolean}
*/
var isAnagram = function(s, t) {
var sArr = s.split('');
var tArr = t.split('');

sArr.sort();
tArr.sort();

return sArr.join('')===tArr.join('');
};
