var dotIndex = number.indexOf('.')

function toFixed(value, digits) {
// 返回四舍五入后的字符串
const temp = value * Math.pow(10, digits);
let [left, right] = String(temp).split('.');

if (Number(right) >= 5) {
left = Number(left) + 1;
}

const leftArr = String(left).split('');
leftArr.splice(-digits, 0, '.');
return leftArr.join('');
}
